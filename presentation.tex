\section{Motivation}
% ============================================================

\begin{frame}{Motivation}
We formalize a simple file system as a development graph in Alonzo.

\medskip
A file system naturally involves:
\begin{itemize}
  \item files as named content-bearing objects,
  \item permissions attached to files,
  \item directories containing files,
  \item operations on directories such as containment and size.
\end{itemize}

\medskip
Our goal is to organize these ideas modularly using:
\begin{itemize}
  \item theory definitions,
  \item theory extensions,
  \item a development definition,
  \item and a theory morphism.
\end{itemize}
\end{frame}

\begin{frame}{Why Alonzo?}
Alonzo is well suited for this project because it supports:
\begin{itemize}
  \item higher-order formalization in simple type theory,
  \item abstract theories and theory extensions,
  \item developments with definitions and theorems,
  \item theory morphisms in the little theories method.
\end{itemize}

\medskip
Our project uses six modules:
\[
\PERM,\quad \FILETYPE,\quad \SECFILE,\quad \DIR,\quad \DIRDEV,
\quad \mName{FILE\text{-}to\text{-}DIR}.
\]
\end{frame}

% ============================================================
\section{Permissions and Files}
% ============================================================

\begin{frame}{Theory for Permissions}

\textbf{Idea:} model permissions as a finite enumeration inductive type.

\medskip
\medskip

\textbf{Name:} $\PERM$

\textbf{Base type:}  $P$

\textbf{Constants:}
$
\mName{noaccess}_P,\quad
\mName{rd}_P,\quad
\mName{wr}_P,\quad
\mName{admin}_P
$

\textbf{Axioms:}
\begin{enumerate}
  \item $\cDistinctL{\mName{noaccess},\, \mName{rd},\, \mName{wr},\, \mName{admin}}$
  \hfill (all constructors are distinct).
  \item $\cForallX{p}{\cSetTy{P}}{
      \cImpliesX{p\;\mName{noaccess} \wedge p\;\mName{rd} \wedge p\;\mName{wr} \wedge p\;\mName{admin}}{
        \cForallX{x}{P}{p\;x}}}$
        \hfill (structural induction).
\end{enumerate}

\medskip
So $P$ has exactly four elements in every standard model.
\end{frame}

\begin{frame}{Theory for Files}

\textbf{Idea:} a file is a named container for content.

\medskip
\textbf{Name:} $\FILETYPE$

\textbf{Base types:} $N,\quad C,\quad F$

\textbf{Constants:}
$\mkfile_{N \to C \to F},\qquad
\fname_{F \to N},\qquad
\fcontent_{F \to C}$

\textbf{Axioms:}
\begin{enumerate}
    \item $\cTotalB{\mkfile}$
    \hfill ($\mkfile$ is total).
    \item $\cInjB{\mkfile}$
    \hfill ($\mkfile$ is injective).
    \item $\cForallBX{n}{N}{c}{C}{
      \cEqX{\fname\;(\mkfile\;n\;c)}{n}}$
    \hfill (name accessor).
    \item $\cForallBX{n}{N}{c}{C}{
      \cEqX{\fcontent\;(\mkfile\;n\;c)}{c}}$
    \hfill (content accessor).
    \item $\cTotal{\fname}$
    \hfill ($\fname$ is total).
    \item $\cTotal{\fcontent}$
    \hfill ($\fcontent$ is total).
    \item $\cForallX{f}{F}{\cEqX{\mkfile\;(\fname\;f)\;(\fcontent\;f)}{f}}$
    \hfill (surjectivity of $\mkfile$).
\end{enumerate}

\medskip
Thus $F$ behaves like an abstract record type isomorphic to $N \times C$.
\end{frame}

% ============================================================
\section{Secured Files and Directories}
% ============================================================

\begin{frame}{ Theory Extension for Secured Files}

\textbf{Idea:} combine files and permissions.  %$\SECFILE$ is a
%\emph{joint extension} of both $\FILETYPE$ and $\PERM$
%(pushout-style construction).

\medskip
\textbf{Name:} $\SECFILE$

\textbf{Extends:} $\FILETYPE$ and $\PERM$

\medskip
\textbf{New base types:} \emph{None.}

($P$ is inherited from $\PERM$; $N$, $C$, $F$ from $\FILETYPE$).

\medskip
\textbf{New constants:}
$\perm_{F \to P},\qquad
\prot_{F \to P \to F}$

\textbf{New axioms:}
\begin{enumerate}\setcounter{enumi}{7}
    \item $\cTotal{\perm}$
    \hfill ($\perm$ is total).
    \item $\cTotalB{\prot}$
    \hfill ($\prot$ is total).
    \item $\cForallBX{f}{F}{p}{P}{
        \cEqX{\perm\;(\prot\;f\;p)}{p}}$
    \hfill (protect sets permission).
    \item $\cForallBX{f}{F}{p}{P}{
      \cEqX{\fname\;(\prot\;f\;p)}{\fname\;f}}$
    \hfill (protect preserves name).
    \item $\cForallBX{f}{F}{p}{P}{
      \cEqX{\fcontent\;(\prot\;f\;p)}{\fcontent\;f}}$
    \hfill (protect preserves content).
\end{enumerate}
\end{frame}

\begin{frame}{Theory Extension for Directories}

\textbf{Design choice:} directories are modelled as a \emph{flat} structure.

\medskip
\textbf{Name:} $\DIR$

\textbf{Extends:} $\SECFILE$.

\textbf{New base type:} $D$

\textbf{New Constants:}
$\mName{emptydir}_D,\qquad
\mName{addfile}_{F \to D \to D}$

\textbf{New axioms:}
\begin{enumerate}\setcounter{enumi}{12}
    \item $\cTotalB{\mName{addfile}}$
      \hfill ($\mName{addfile}$ is total).
    \item $\cInjB{\mName{addfile}}$
      \hfill ($\mName{addfile}$ is injective).
    \item $\cForallBX{f}{F}{d}{D}{
      \cNotEqX{\mName{addfile}\;f\;d}{\mName{emptydir}}}$
      \hfill (disjointness).
    \item $\cForallX{p}{\cSetTy{D}}{
      \cImpliesX{p\;\mName{emptydir} \;\wedge\;
        \cForallBX{f}{F}{d}{D}{
          \cImpliesX{p\;d}{p\;(\mName{addfile}\;f\;d)}}}{
        \cForallX{d}{D}{p\;d}}}$
      \hfill (structural induction).
\end{enumerate}

\medskip
This is list-like: a directory is either empty or formed by adding a file.
\end{frame}

\begin{frame}{Directories as an Inductive Type}
The constructor set for $D$ is
\[
K_D = \left\{
\mName{emptydir}_D,\;
(\mName{addfile}_{F \to D \to D},\, U_{\cSetTy{F}})
\right\}.
\]

\medskip
Interpretation:
\begin{itemize}
  \item $\mName{emptydir}$ is the nullary constructor,
  \item $\mName{addfile}$ is the recursive constructor,
  \item $D$ is therefore a recursive inductive type.
\end{itemize}

\medskip
In every standard model, $D$ is isomorphic to the set of
finite lists of elements of $D_F^M$.
\end{frame}

\section{Directory Development}
% ============================================================

\begin{frame}{Directory Development}
\small
We now define derived operations on directories and prove key
properties.
\medskip

\textbf{Name:} $\DIRDEV$

\textbf{Bottom theory:} $\DIR$.

\medskip
\textbf{Definitions and theorems:}

\medskip
\textbf{Def\,1} ($\mName{contains}$). \quad
$\mName{contains}_{\cFunTyBX{N}{D}{\cB}} \;=\;
\cDefDesX{g}{N \to D \to \cB}{
  \cForallX{n}{N}{\cEqX{g\;n\;\mName{emptydir}}{\cF}} \;\wedge\;
  \cForallCX{n}{N}{f}{F}{d}{D}{
    \cEqX{g\;n\;(\mName{addfile}\;f\;d)}{
      (\cEqX{\fname\;f}{n}) \vee g\;n\;d}}}$

\medskip
\textbf{Thm\,1.} \quad $\cTotalB{\mName{contains}}$
\hfill ($\mName{contains}$ is total).

\textbf{Thm\,2.} \quad
$\cForallX{n}{N}{\cEqX{\mName{contains}\;n\;\mName{emptydir}}{\cF}}$
\hfill (empty directory contains nothing).

\medskip
\textbf{Def\,2} ($\mName{size}$). \quad
$\mName{size}_{\cFunTyX{D}{\mathbb{N}}} \;=\;
\cDefDesX{g}{D \to \mathbb{N}}{
  \cEqX{g\;\mName{emptydir}}{0} \;\wedge\;
  \cForallBX{f}{F}{d}{D}{
    \cEqX{g\;(\mName{addfile}\;f\;d)}{g\;d + 1}}}$

\medskip
\textbf{Thm\,3.} \quad $\cTotal{\mName{size}}$
\hfill ($\mName{size}$ is total).

\textbf{Thm\,4.} \quad
$\cEqX{\mName{size}\;\mName{emptydir}}{0}$
\hfill (empty directory has size zero).

\textbf{Thm\,5.} \quad
$\cForallBX{f}{F}{d}{D}{
  \cEqX{\mName{size}\;(\mName{addfile}\;f\;d)}{\mName{size}\;d + 1}}$
\hfill (addfile increments size).

\textbf{Thm\,6.} \quad
$\cForallX{d}{D}{\mName{size}\;d \ge 0}$
\hfill (size is non-negative).

\end{frame}

\begin{frame}{Explanation for Directory Development}
\small
\textbf{Contains.}
The predicate $\mName{contains}\;n\;d$ tests whether a file with
name $n$ exists in directory $d$. It is defined by structural
recursion on $d$:
\begin{itemize}
  \item the empty directory contains nothing;
  \item $\mName{addfile}\;f\;d$ contains $n$ iff either
  $\fname\;f = n$ or $d$ already contains $n$.
\end{itemize}

\medskip
\textbf{Size.}
The function $\mName{size}\;d$ returns the number of file entries
in directory $d$:
\begin{itemize}
  \item $\mName{size}\;\mName{emptydir} = 0$;
  \item $\mName{size}\;(\mName{addfile}\;f\;d) = \mName{size}\;d + 1$.
\end{itemize}

\medskip
Both definitions are validated in the appendix by proving that the
corresponding definite descriptions denote values.
\end{frame}

\begin{frame}{Example of Directory Evaluation}
\small
Let
\[
f_1 \;=\; \prot\;(\mkfile\;n_1\;c_1)\;\mName{rd},
\qquad
f_2 \;=\; \prot\;(\mkfile\;n_2\;c_2)\;\mName{wr},
\]
and define
\[
d \;=\; \mName{addfile}\;f_1\;(\mName{addfile}\;f_2\;\mName{emptydir}).
\]

Then:
\[
\mName{contains}\;n_1\;d = \cT,
\qquad
\mName{contains}\;n_2\;d = \cT,
\]
\[
\mName{contains}\;n_3\;d = \cF,
\qquad
\mName{size}\;d = 2.
\]

\medskip
Thus $d$ is a flat directory with exactly two file entries.
\end{frame}

\section{Theory Morphism}
% ============================================================

\begin{frame}[fragile]{Theory Translation}
\small

To obtain a well-formed theory translation, we map
it to the closed quasitype of single-entry directories:
\[
  \zeta_F \;=\;
  \{\, d : D \mid \cForsomeX{f}{F}{
      \cEqX{d}{\mName{addfile}\;f\;\mName{emptydir}}}\,\}.
\]

\medskip

For each $d \in \zeta_F$, define:
\[
  \mName{extract}(d) \;=\;
  \cDefDesX{f}{F}{
    \cEqX{d}{\mName{addfile}\;f\;\mName{emptydir}}}.
\]

\medskip

\end{frame}

\begin{frame}[fragile]{Theory Translation (Cont'd)}
\small

\textbf{Name:} $\mName{FILE\text{-}to\text{-}DIR}$

\textbf{Source theory:} $\FILETYPE$

\textbf{Target theory:} $\DIR$

\textbf{Base type mapping:}
\begin{itemize}
  \item[] $N \;\mapsto\; N$ \quad (names preserved)
  \item[] $C \;\mapsto\; C$ \quad (contents preserved)
  \item[] $F \;\mapsto\; \zeta_F$
\end{itemize}

\textbf{Constants mapping:}
\begin{itemize}
  \item[] $\mkfile_{N \to C \to F} \;\mapsto\;
    \cFunAbsX{n}{N}{\cFunAbsX{c}{C}{
      \mName{addfile}\;(\mkfile\;n\;c)\;\mName{emptydir}}}$
  \item[] $\fname_{F \to N} \;\mapsto\;
    \cFunAbsX{d}{D}{\fname\;(\mName{extract}(d))}$
  \item[] $\fcontent_{F \to C} \;\mapsto\;
    \cFunAbsX{d}{D}{\fcontent\;(\mName{extract}(d))}$
\end{itemize}

\end{frame}

\begin{frame}{Verification of the Morphism}
\small

Key observation: if $d \in \zeta_F$, then $d$ has the form
$\mName{addfile}\;f\;\mName{emptydir}$ for a unique $f : F$.
Hence
\[
\mName{extract}\;(\mName{addfile}\;f\;\mName{emptydir}) = f
\]
by uniqueness of the definite description and injectivity of
$\mName{addfile}$.

\medskip
Therefore:
\begin{itemize}
  \item TOTAL2($\mkfile$) maps to totality of the translated constructor;
  \item INJ2($\mkfile$) maps to injectivity;
  \item the accessor axioms for $\fname$ and $\fcontent$ are preserved;
  \item totality of the translated accessors holds on $\zeta_F$;
  \item surjectivity is preserved on the embedded image.
\end{itemize}

\medskip
Thus the translation is well formed. By the Morphism Theorem,
theorems of $\FILETYPE$ transport to $\DIR$ under
$\mName{FILE\text{-}to\text{-}DIR}$.
\end{frame}

\section{Conclusion}
% ============================================================

\begin{frame}{Conclusion}
\small
We formalized a simple file system in Alonzo using six modules:
\[
\PERM,\quad
\FILETYPE,\quad
\SECFILE,\quad
\DIR,\quad
\DIRDEV,\quad
\mName{FILE\text{-}to\text{-}DIR}.
\]

\medskip
Main outcomes:
\begin{itemize}
  \item permissions and files are specified independently;
  \item secured files and flat directories are built modularly;
  \item directory operations are defined in a development;
  \item files are embedded into directories by a theory morphism.
\end{itemize}
\end{frame}
